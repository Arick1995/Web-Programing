```c#
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

public partial class Default2 : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e)
    {
        Label3.Text = Request.QueryString["a"];
        if (Session["v"] == 0 || Session["v"] == null )
        {
            Response.Redirect("Default.aspx");
        }
    }

    protected void Button1_Click(object sender, EventArgs e)
    {
       
        try
        {
            BMI b = new BMI();
            double BMI_Value;
            double h = Double.Parse(TextBox1.Text);
            double w = Double.Parse(TextBox2.Text);
            if (RadioButtonList1.SelectedValue == "0")
            {
                b.height = h / 100;
            }
            else
            {
                b.height = h * 2.54 / 100;
            }
            if (RadioButtonList2.SelectedValue == "0")
            {
                b.weight = w;
            }
            else
            {
                b.weight = w * 0.4536;
            }
            BMI_Value = b.BMI_calulate();
            Label1.Text = Math.Round(BMI_Value, 3).ToString();
            Label2.Text = b.BMI_Desc(BMI_Value);
        }
        catch (Exception err)
        {
            Label1.Text = err.Message;
        }
       
    }
}
```

## BMI.cs
```c#
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

/// <summary>
/// Class1 的摘要描述
/// </summary>
public class BMI
{
    public BMI()
    {
        //
        // TODO: 在這裡新增建構函式邏輯
        //
    }

    public double height
    {
        get;
        set;
    }

    public double weight
    {
        get;
        set;
    }

    public double BMI_calulate()
    {
        return 1.3 * weight / (Math.Pow(height, 2.5));
    }
    public string BMI_Desc(double i)
    {
        if (i < 15)
        {
            return "Starvation";
        } else if (i > 15 && i < 18.5)
        {
            return "Underweight";
        } else if (i >= 18.5 && i < 25)
        {
            return "Normal";
        } else if (i >= 25 && i < 30)
        {
            return "Oberweight";
        } else if (i >= 30 && i < 40)
        {
            return "Obese";
        }
        else
        {
            return "Morbidlv Obese";
        }
    }
}
```
